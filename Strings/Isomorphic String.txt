class Solution {
public:
    bool isIsomorphic(string s, string t) {
        vector<int> s_map(256,0);
        vector<int> t_map(256,0);

        int n = s.length();

        for(int i = 0 ; i<n ; i++)
        {
            if(s_map[s[i]] != t_map[t[i]])
                return false;
            s_map[s[i]] = i+1;
            t_map[t[i]] = i+1;
        }
        return true;
    }
};
