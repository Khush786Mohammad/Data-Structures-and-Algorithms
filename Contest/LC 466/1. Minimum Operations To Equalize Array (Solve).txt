class Solution {
public:
    int minOperations(vector<int>& nums) {
        int n = nums.size();

        int value = 0;

        for(int i = 0 ; i<n-1 ; i++)
            {
                if(nums[i] != nums[i+1])
                {
                    value = 1;
                    break;
                }
            }

        if(value == 0)
            return 0;
        else
            return 1;
    }
};
