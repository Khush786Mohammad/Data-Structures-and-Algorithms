
class Solution {
    // Function to find the leaders in the array.
  public:
    vector<int> leaders(vector<int>& arr) {
        vector<int> ans;
        int currMax = -1;
        int n = arr.size();
        
        for(int i = n-1 ; i>=0 ; i--)
        {
            if(arr[i] >= currMax)
            {
                ans.push_back(arr[i]);
                currMax = arr[i];
            }
        }
        reverse(ans.begin(), ans.end());
        return ans;
    }
};

T.C = O(N);
S.C = O(1)
