class Solution {
public:
    int subarraySum(vector<int>& nums, int k) {
        int n = nums.size();
        int ans = 0;
        int prefixSum = 0;
        // [prefixSum , count]
        unordered_map<int,int> mp;
        mp[0] = 1;

        for(int i = 0 ; i<n ; i++){
            prefixSum += nums[i];
            if(mp.find(prefixSum - k) != 0){
                ans += mp[prefixSum-k];
            }
            mp[prefixSum] += 1;
        }
        return ans;
    }
};
