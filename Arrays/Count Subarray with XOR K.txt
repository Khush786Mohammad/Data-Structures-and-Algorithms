class Solution {
  public:
    long subarrayXor(vector<int> &arr, int k) {
        int n = arr.size();
        int ans = 0;
        int XOR = 0;
        unordered_map<int,int> mp;
        
        for(int i = 0 ; i<n ; i++){
            XOR ^= arr[i];
            int remaining = XOR ^ k;
            if(XOR == k)
                ans++;
            if(mp.find(remaining) != mp.end())
                ans += mp[remaining];
            mp[XOR] += 1;
        }
        return ans;
    }
};
