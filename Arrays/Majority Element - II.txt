class Solution {
public:
    vector<int> majorityElement(vector<int>& nums) {
        int n = nums.size();
        vector<int> ans;

        int cnt1 = 0;
        int cnt2 = 0;
        int el1 = INT_MIN;
        int el2 = INT_MIN;

        for(int i = 0 ; i<n ; i++){
            if(cnt1 == 0 && nums[i] != el2){
                cnt1 = 1;
                el1 = nums[i];
            }
            else if(cnt2 == 0 && nums[i] != el1){
                cnt2 = 1;
                el2 = nums[i];
            }
            else if(nums[i] == el1)
                cnt1++;
            else if(nums[i] == el2)
                cnt2++;
            else{
                cnt1--;
                cnt2--;
            }
        }

        int freq1 = 0;
        int freq2 = 0;

        for(int i = 0 ; i<n ; i++){
            if(nums[i] == el1)
                freq1++;
            else if(nums[i] == el2)
                freq2++;
        }
        if(freq1 > n/3)
            ans.push_back(el1);
        if(freq2 > n/3)
            ans.push_back(el2);
        return ans;
    }
};
